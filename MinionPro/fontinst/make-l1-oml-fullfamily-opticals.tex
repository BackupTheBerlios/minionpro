\def\installopt#1#2#3{%
  \installfont{MinionPro-MediumMixed#3#1-l1-oml}{%
     MnSymbolA#2                       encoding mn-symbol-a,%
     MnSymbolC#2                       encoding mn-symbol-c,%
     MnSymbolD#2                       encoding mn-symbol-d,%
     MnSymbolE#2                       encoding mn-symbol-e,%
     MinionPro-Medium#1-tosf-oml       encoding mn-oml-up,%
     MinionPro-MediumIt#1-tosf-oml     encoding mn-oml-it,%
     haxmnoml,%
     haxmnoml-MediumMixed#1,%
     arrowhooks%
  }{mn-oml#3}{}{}{}{}{}%
}

\installfonts
\setint{letterspacing}{80}
  \installopt{Capt}{7}{}
  \installopt{Subh}{12}{}
  \installopt{Disp}{12}{}
  \installopt{Capt}{7}{Up}
  \installopt{Subh}{12}{Up}
  \installopt{Disp}{12}{Up}
\endinstallfonts

\bye
