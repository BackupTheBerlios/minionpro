\input fontinst-macros.tex

\def\installopt#1{%
  \installfont{MinionPro-MediumMixed#1-oml}{%
     MnSymbolA10                       encoding mn-symbol-a,%
     MnSymbolC10                       encoding mn-symbol-c,%
     MnSymbolD10                       encoding mn-symbol-d,%
     MnSymbolE10                       encoding mn-symbol-e,%
     MinionPro-Medium-tosf-oml         encoding mn-oml-up,%
     MinionPro-MediumIt-tosf-oml       encoding mn-oml-it,%
     adjust-oml,%
     numfracs,%
     arrowhooks%
  }{mn-oml#1}{}{}{}{}{}%
}

\installfonts
  \setint{letterspacing}{40}
  \installopt{}
  \installopt{Up}
\endinstallfonts

\bye
