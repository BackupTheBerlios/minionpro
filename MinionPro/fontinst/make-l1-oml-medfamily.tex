\input fontinst-macros.tex

\def\installopt#1{%
  \installfont{MinionPro-SemiboldMixed#1-l1-oml}{%
     MnSymbolA-Bold10                  encoding mn-symbol-a,%
     MnSymbolC-Bold10                  encoding mn-symbol-c,%
     MnSymbolD-Bold10                  encoding mn-symbol-d,%
     MnSymbolE-Bold10                  encoding mn-symbol-e,%
     MinionPro-Semibold-tosf-oml       encoding mn-oml-up,%
     MinionPro-SemiboldIt-tosf-oml     encoding mn-oml-it,%
     adjust-oml,%
     numfracs,%
     arrowhooks%
  }{mn-oml#1}{}{}{}{}{}%
}

\installfonts
  \setint{letterspacing}{80}
  \installopt{}
  \installopt{Up}
\endinstallfonts

\bye
