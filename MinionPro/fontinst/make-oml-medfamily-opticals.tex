\def\installopt#1#2#3{%
  \installfont{MinionPro-Mixed#3#1-oml}{%
     MnSymbolA#2                       encoding mn-symbol-a,%
     MnSymbolC#2                       encoding mn-symbol-c,%
     MnSymbolD#2                       encoding mn-symbol-d,%
     MnSymbolE#2                       encoding mn-symbol-e,%
     MinionPro-#1-tosf-oml             encoding mn-oml-up,%
     MinionPro-It#1-tosf-oml           encoding mn-oml-it,%
     adjust-oml,%
     arrowhooks%
  }{mn-oml#3}{}{}{}{}{}%
  \installfont{MinionPro-SemiboldMixed#3#1-oml}{%
     MnSymbolA-Bold#2                  encoding mn-symbol-a,%
     MnSymbolC-Bold#2                  encoding mn-symbol-c,%
     MnSymbolD-Bold#2                  encoding mn-symbol-d,%
     MnSymbolE-Bold#2                  encoding mn-symbol-e,%
     MinionPro-Semibold#1-tosf-oml     encoding mn-oml-up,%
     MinionPro-SemiboldIt#1-tosf-oml   encoding mn-oml-it,%
     adjust-oml,%
     arrowhooks%
  }{mn-oml#3}{}{}{}{}{}%
  \installfont{MinionPro-BoldMixed#3#1-oml}{%
     MnSymbolA-Bold#2                  encoding mn-symbol-a,%
     MnSymbolC-Bold#2                  encoding mn-symbol-c,%
     MnSymbolD-Bold#2                  encoding mn-symbol-d,%
     MnSymbolE-Bold#2                  encoding mn-symbol-e,%
     MinionPro-Bold#1-tosf-oml         encoding mn-oml-up,%
     MinionPro-BoldIt#1-tosf-oml       encoding mn-oml-it,%
     adjust-oml,%
     arrowhooks%
  }{mn-oml#3}{}{}{}{}{}%
}

\installfonts
\setint{letterspacing}{40}
  \installopt{Capt}{7}{}
  \installopt{Subh}{12}{}
  \installopt{Disp}{12}{}
  \installopt{Capt}{7}{Up}
  \installopt{Subh}{12}{Up}
  \installopt{Disp}{12}{Up}  
\endinstallfonts

\bye
