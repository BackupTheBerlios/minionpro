\def\installopt#1#2{%
  \installfont{MinionPro-MediumMixed#1-oml}{%
     MnSymbolA#2                       encoding mn-symbol-a,%
     MnSymbolC#2                       encoding mn-symbol-c,%
     MnSymbolD#2                       encoding mn-symbol-d,%
     MnSymbolE#2                       encoding mn-symbol-e,%
     MinionPro-Medium#1-tosf-oml       encoding mn-oml-up,%
     MinionPro-MediumIt#1-tosf-oml     encoding mn-oml-it,%
     haxmnoml-MediumMixed#1,%
     arrowhooks%
  }{mn-oml}{}{}{}{}{}%
}

\installfonts
\setint{letterspacing}{40}
  \installopt{Capt}{7}
  \installopt{Subh}{12}
  \installopt{Disp}{12}
\endinstallfonts

\bye
