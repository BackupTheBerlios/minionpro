\input fontinst-macros.tex

\def\installopt#1{%
  \installfont{MinionPro-Mixed#1-l1-oml}{%
     MnSymbolA10                       encoding mn-symbol-a,%
     MnSymbolC10                       encoding mn-symbol-c,%
     MnSymbolD10                       encoding mn-symbol-d,%
     MnSymbolE10                       encoding mn-symbol-e,%
     MinionPro-Regular-tosf-oml        encoding mn-oml-up,%
     MinionPro-It-tosf-oml             encoding mn-oml-it,%
     adjust-oml,%
     numfracs,%
     arrowhooks%
  }{mn-oml#1}{}{}{}{}{}%
  \installfont{MinionPro-BoldMixed#1-l1-oml}{%
     MnSymbolA-Bold10                  encoding mn-symbol-a,%
     MnSymbolC-Bold10                  encoding mn-symbol-c,%
     MnSymbolD-Bold10                  encoding mn-symbol-d,%
     MnSymbolE-Bold10                  encoding mn-symbol-e,%
     MinionPro-Bold-tosf-oml           encoding mn-oml-up,%
     MinionPro-BoldIt-tosf-oml         encoding mn-oml-it,%
     adjust-oml,%
     numfracs,%
     arrowhooks%
  }{mn-oml#1}{}{}{}{}{}%
}

\installfonts
  \setint{letterspacing}{80}
  \installopt{}
  \installopt{Up}
\endinstallfonts

\bye
