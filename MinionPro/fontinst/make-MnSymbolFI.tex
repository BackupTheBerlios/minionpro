\def\metrics_given_encoding#1#2{
   \x_cs\edef{s-afm-name}{#1-#2}
   \if_file_exists{#1.pl}\then
      \generalpltomtx{#1}{#1-#2}{pl}{#2}
   \else
      \if_file_exists{#1.vpl}\then
         \generalpltomtx{#1}{#1-#2}{vpl}{#2}
      \else
         \unsetstr{afm-name}
   \fi\fi
}

\def\installopt#1#2{%
  \installfont{#1}{%
     MnSymbolF10                       encoding mn-symbol-f-partialints,%
     #2                                encoding mn-symbol-f,%
     MinionPro-It-osf-oml--base        encoding mn-tint scaled 1100,%
     MinionPro-ItDisp-osf-cn1-integral encoding mn-dint scaled 1100,%
     mn-intop-text,%
     mn-intop-disp%
  }{mn-symbol-fi}{}{}{}{}{}%
}

\def\installboldopt#1#2{%

     MnSymbolF-Bold10                      encoding mn-symbol-f-partialints,%
     #2                                    encoding mn-symbol-f,%
     MinionPro-BoldIt-osf-oml--base        encoding mn-tint scaled 1140,%
     MinionPro-BoldItDisp-osf-cn1-integral encoding mn-dint scaled 1140,%
     mn-intop-bold-text,%
     mn-intop-bold-disp%
  }{mn-symbol-fi}{}{}{}{}{}%
}
